# Take a sentence as input
sentence = input("Enter a sentence: ")

vowels = "aeiouAEIOU"
v_count = c_count = d_count = s_count = 0

# a) Count vowels, consonants, digits, and special characters.
for ch in sentence:
    if ch.isdigit():
        d_count += 1
    elif ch.isalpha():
        if ch in vowels:
            v_count += 1
        else:
            c_count += 1
    elif ch != " ":
        s_count += 1

print("Vowels:", v_count)
print("Consonants:", c_count)
print("Digits:", d_count)
print("Special Characters:", s_count)

# b) Print each word in reverse using a for loop.
words = sentence.split()
print("\nEach word in reverse:")
for w in words:
    print(w[::-1])

# c) Print the longest word from the sentence.
longest = ""
for w in words:
    if len(w) > len(longest):
        longest = w

print("\nLongest word:", longest)
